\section*{Aufgabe 5.2}
\begin{minted}{haskell}
    sumr [1, 2, 3, 4]
\end{minted}
wird folgendermaßen ausgeführt:
\begin{align*}
    4 &+ 0 & [1, 2, 3, 4] \\
    3 &+ 4 & [1, 2, 3] \\
    2 &+ 7 & [1, 2] \\
    1 &+ 9 & [1] \\
    10
\end{align*}

während
\begin{minted}{haskell}
    suml [1, 2, 3, 4]
\end{minted}
wie folgt ausgeführt wird:
\begin{align*}
    0 &+ 1 & [1, 2, 3, 4] \\
    1 &+ 2 & [2, 3, 4] \\
    3 &+ 3 & [3, 4] \\
    6 &+ 4 & [4] \\
    10
\end{align*}


\section*{Aufgabe 5.3}
\paragraph{1)}
\begin{minted}{haskell}
foldFor :: (Enum b) => a -> b -> b -> (a -> b -> a) -> a
foldFor x0 i0 i1 f = foldl f x0 [i0 .. i1]
\end{minted}

\paragraph{2)}
\begin{minted}{haskell}
foldWhile :: (Enum b) => (a -> b -> a) -> (b -> Bool) -> a -> b -> a
foldWhile f a x0 i0 = foldl f x0 $ takeWhile a [i0..]
\end{minted}
