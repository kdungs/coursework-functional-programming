\section*{Aufgabe 4.3}
\begin{minted}{haskell}
    collatz :: Int -> Int
    collatz n
        | n `mod` 2 == 0 = n `div` 2
        | otherwise      = 3 * n + 1

    collatzSequence :: Int -> [Int]
    collatzSequence n = takeWhile (/=1) (iterate collatz n)
\end{minted}
